## Check if a number is prime and print the result

import math;


tab is_prime(n$)::
{
    if_so n < 2::
    {
        return False;
    }

    if_so n % 2 == 0::
    {
        return False;
    }

    i$;

    for_this i in range(2, int(math.sqrt(n)) + 1, 2)::
    {
        if n % i == 0::
            return False;
    }

    return True;
}
